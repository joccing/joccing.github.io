"use strict";

/**
 * 
 * Initialize the table structure.  This needs to be done after it is defined.
 *
 * @param {ctTable} table - The variable containing a CachedTable.
 * @returns {Object} The ctTable functions.
 */
function ctInit( table ) {
  return table.init();
}

/**
 * 
 * Given a CachedTable, this is an iterator that returns the next row of data.
 *
 * @param {ctTable} table - The variable containing a CachedTable.
 * @returns {Object[]} Returns null after the last row or an array of Object.
 */
 function ctNextRow( table ){
   return table.nextRow();
 }

/**
 * Returns the current row number based on the starting row number stated
 * at creation.  If the starting row was 10, then this return value will be
 * at least 10.
 *
 * @param {ctTable} table - The variable containing a CachedTable.
 * @returns {integer} Value that is based on the starting row number.
 */ 
function ctGetRowIndex( table ){
  return table.getRowIndex();
}

/**
 * Returns a single row specified by its row number.
 *
 * @param {ctTable} table - The variable containing a CachedTable.
 * @param {integer} row - Value based on the starting row number at creation.
 * @returns {Object[]} Throws an exception if row value exceeds ending row number specified at creation.
 * @throws {Error} if row value exceeds ending row number specified at creation.
 */
function ctGetRow( table, row ){
  return table.getRow(row);
}

/**
 * Set the values for a row in the CachedTable.  By default this function
 * accepts an array of values, but for convenience, it will also accept a single array element of an array
 * which is the actual input argument used by the SpreadsheetApp service.
 *
 * @param {ctTable} table - The variable containing a CachedTable.
 * @param {integer} row - Value based on the starting row number at creation.
 * @param {Object[]} values - An array of values for the row.  
 * @throws {Error} if row value exceeds ending row number specified at creation.
 * @throws {Error} if number of values does not fit number of columns defined at creation.
 * @returns {Object} The ctTable functions.
 */
function ctSetRow( table, row, values ){
  return table.setRow( row, values );
}

/**
 * Set the value for a cell in the CachedTable specified by its row and column number.
 *
 * @param {ctTable} table - The variable containing a CachedTable.
 * @param {integer} row - Value based on the starting row number at creation.
 * @param {integer} col - Value based on the starting column number at creation.
 * @param {Object} value - An value for the cell.  
 * @throws {Error} if row or col value exceeds table bounds specified at creation.
 * @returns {Object} The ctTable functions.
 */
function ctSetValue( table, row, col, value ){
  return table.setValue( row, col, value );
}

/**
 * Get the value for a cell in the CachedTable specified by its row and column number.
 *
 * @param {ctTable} table - The variable containing a CachedTable.
 * @param {integer} row - Value based on the starting row number at creation.
 * @param {integer} col - Value based on the starting column number at creation.
 * @returns {Object} Value that was stored in the cell.
 * @throws {Error} if row or col value exceeds table bounds specified at creation.
 */
function ctGetValue( table, row, col, value ){
  return table.getValue( row, col, value );
}

/**
 * Get the number of rows in the CachedTable.
 *
 * @param {ctTable} table - The variable containing a CachedTable.
 * @returns {integer} Number of rows in the CachedTable.
 */
function ctRows( table ){
  return table.rows();
}

/**
 * Get the number of columns in the CachedTable.
 *
 * @param {ctTable} table - The variable containing a CachedTable.
 * @returns {integer} Number of columns in the CachedTable.
 */
function ctColumns( table ){
  return table.columns();
}

/**
 * Write contents of cache back to Google sheet.
 *
 * @param {ctTable} table - The variable containing a CachedTable.
 */
function ctFlushAll( table ){
  return table.flushAll();
}
