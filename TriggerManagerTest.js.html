// Maintain cell colour and alignment for CGP links
function MaintainCellColour( e, row, col ){
          
  e.range.setBackground("#fff2cc");
  e.range.setFontFamily("Open Sans");
  e.range.setFontSize(12);
  e.range.setHorizontalAlignment("left");
  e.range.setWrapStrategy(SpreadsheetApp.WrapStrategy.CLIP);
}

// Maintain cell formation using another one as example
function MaintainCellLike( e, row, col, args ){

  var cell = SpreadsheetApp.getActiveSheet().getRange(args[0]);
  cell.copyFormatToRange(
    e.range.getSheet(), 
    e.range.getColumn(), 
    e.range.getLastColumn(), 
    e.range.getRow(), 
    e.range.getLastRow());
}
